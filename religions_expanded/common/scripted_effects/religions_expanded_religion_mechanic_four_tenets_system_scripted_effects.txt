re_set_religion_mechanic_four_tenets_system_value = {
	if = {
		limit = {
			re_uses_religion_mechanic = { mechanic = four_tenets_system }
		}
		[[value]
			set_variable = { which = re_religion_mechanic_four_tenets_system_value value = $value$ }
		]
		[[which]
			set_variable = { which = re_religion_mechanic_four_tenets_system_value which = $which$ }
		]
		hidden_effect = {
			if = {
				limit = {
					re_has_religion_mechanic_four_tenets_system_value = { value = 100 }
				}
				set_variable = { which = re_religion_mechanic_four_tenets_system_value value = 100 }
			}
			else_if = {
				limit = {
					NOT = { re_has_religion_mechanic_four_tenets_system_value = { value = 0 } }
				}
				set_variable = { which = re_religion_mechanic_four_tenets_system_value value = 0 }
			}
		}
	}
}

re_export_monthly_religion_mechanic_four_tenets_system_value_gain = {
	hidden_effect = {
		export_to_variable = {
			which = re_monthly_religion_mechanic_four_tenets_system_value
			value = modifier:re_monthly_religion_mechanic_four_tenets_system_change
		}
		if = {
			limit = {
				has_dlc = "Rule Britannia"
			}
			change_variable = { which = re_monthly_religion_mechanic_four_tenets_system_value value = 0.2 } # Base Gain
		}
		else = {
			change_variable = { which = re_monthly_religion_mechanic_four_tenets_system_value value = 0.6 } # Base Gain (Higher cause no Innovativeness)
		}
		set_variable = { which = re_monthly_religion_mechanic_four_tenets_system_value_base which = re_monthly_religion_mechanic_four_tenets_system_value } # display
		# Modifier
		export_to_variable = {
			which = re_monthly_religion_mechanic_four_tenets_system_modifier_value
			value = modifier:re_monthly_religion_mechanic_four_tenets_system_change_modifier
		}
		if = {
			limit = {
				check_variable = { which = re_monthly_religion_mechanic_four_tenets_system_value value = 0 }
			}
			if = {
				limit = {
					NOT = { check_variable = { which = re_monthly_religion_mechanic_four_tenets_system_modifier_value value = -1 } }
				}
				set_variable = { which = re_monthly_religion_mechanic_four_tenets_system_modifier_value value = -1 }
			}
			set_variable = { which = re_monthly_religion_mechanic_four_tenets_system_modifier_value_base which = re_monthly_religion_mechanic_four_tenets_system_modifier_value } # display
			multiply_variable = { which = re_monthly_religion_mechanic_four_tenets_system_modifier_value_base value = 100 } # display
			change_variable = { which = re_monthly_religion_mechanic_four_tenets_system_modifier_value value = 1 }
		}
		else = {
			if = {
				limit = {
					check_variable = { which = re_monthly_religion_mechanic_four_tenets_system_modifier_value value = 1 }
				}
				set_variable = { which = re_monthly_religion_mechanic_four_tenets_system_modifier_value value = 1 }
			}
			set_variable = { which = re_monthly_religion_mechanic_four_tenets_system_modifier_value_base which = re_monthly_religion_mechanic_four_tenets_system_modifier_value } # display
			multiply_variable = { which = re_monthly_religion_mechanic_four_tenets_system_modifier_value_base value = 100 } # display
			change_variable = { which = re_monthly_religion_mechanic_four_tenets_system_modifier_value value = -1 }
			multiply_variable = { which = re_monthly_religion_mechanic_four_tenets_system_modifier_value value = -1 }
		}
		multiply_variable = { which = re_monthly_religion_mechanic_four_tenets_system_value which = re_monthly_religion_mechanic_four_tenets_system_modifier_value }
		set_variable = { which = re_monthly_religion_mechanic_four_tenets_system_modifier_value value = 0 }
	}
}

re_add_monthly_religion_mechanic_four_tenets_system_value = {
	re_export_monthly_religion_mechanic_four_tenets_system_value_gain = yes
	re_add_religion_mechanic_four_tenets_system_value = { which = re_monthly_religion_mechanic_four_tenets_system_value }
}
