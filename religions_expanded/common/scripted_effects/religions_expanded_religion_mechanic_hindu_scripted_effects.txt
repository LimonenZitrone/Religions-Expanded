re_update_religion_mechanic_hindu_dedication_levels = {
	hidden_effect = {
		re_update_religion_mechanic_hindu_dedication_levels_specific = { major_deity = shiva }
		re_update_religion_mechanic_hindu_dedication_levels_specific = { major_deity = brahma }
		re_update_religion_mechanic_hindu_dedication_levels_specific = { major_deity = vishnu }
		re_update_religion_mechanic_hindu_dynamic_modifier = yes
		re_export_monthly_religion_mechanic_hindu_value_gain = yes
	}
}
re_update_religion_mechanic_hindu_dedication_levels_specific = {
	while = {
		limit = {
			check_variable = { which = re_religion_mechanic_hindu_$major_deity$_value value = 100 }
			OR = {
				NOT = { is_variable_equal = { which = re_religion_mechanic_hindu_$major_deity$_value value = 100 } }
				NOT = { re_has_religion_mechanic_hindu_major_deity_dedication_level = { major_deity = $major_deity$ level = 5 } }
			}
		}
		if = {
			limit = {
				re_has_religion_mechanic_hindu_major_deity_dedication_level = { major_deity = $major_deity$ level = 5 }
			}
			set_variable = { which = re_religion_mechanic_hindu_$major_deity$_value value = 100 }
		}
		else = {
			subtract_variable = { which = re_religion_mechanic_hindu_$major_deity$_value value = 100 }
			change_variable = { which = re_religion_mechanic_hindu_$major_deity$_level value = 1 }
		}
	}
	while = {
		limit = {
			NOT = { check_variable = { which = re_religion_mechanic_hindu_$major_deity$_value value = 0 } }
		}
		if = {
			limit = {
				NOT = { re_has_religion_mechanic_hindu_major_deity_dedication_level = { major_deity = $major_deity$ level = 1 } }
			}
			set_variable = { which = re_religion_mechanic_hindu_$major_deity$_value value = 0 }
		}
		else = {
			change_variable = { which = re_religion_mechanic_hindu_$major_deity$_value value = 100 }
			subtract_variable = { which = re_religion_mechanic_hindu_$major_deity$_level value = 1 }
		}
	}
}

			# Level of Dedication
			if = {
				limit = {
					re_has_religion_mechanic_hindu_major_deity_dedication_level = { major_deity = current level = 5 }
				}
				set_variable = { which = re_monthly_religion_mechanic_hindu_value_dedication_level value = -0.09 }
			}
			else_if = {
				limit = {
					re_has_religion_mechanic_hindu_major_deity_dedication_level = { major_deity = current level = 4 }
				}
				set_variable = { which = re_monthly_religion_mechanic_hindu_value_dedication_level value = -0.07 }

			else_if = {
				limit = {

				}
				set_variable = { which = re_monthly_religion_mechanic_hindu_value_dedication_level value = -0.05 }

				limit = {
					re_has_religion_mechanic_hindu_major_deity_dedication_level = { major_deity = current level = 2 }
				}
				set_variable = { which = re_monthly_religion_mechanic_hindu_value_dedication_level value = -0.03 }

			}
			else_if = {

			}
			else_if = {

					NOT = { religious_unity = 0.75 }
				}

			}
			else = {
				set_variable = { which = re_monthly_religion_mechanic_hindu_value_religious_unity value = 0 }
			}
			change_variable = { which = re_monthly_religion_mechanic_hindu_value which = re_monthly_religion_mechanic_hindu_value_religious_unity }

							ADM = 5

					duration = -1
					hidden = yes
				}
			}
			else_if = {
				limit = {
					re_has_religion_mechanic_hindu_major_deity_dedication_level = { major_deity = vishnu level = 2 }
				}
				add_country_modifier = {
					name = re_religion_mechanic_hindu_modifier_vishnu_buddha_2
					duration = -1
					hidden = yes
				}
			}
			else_if = {
				limit = {
					re_has_religion_mechanic_hindu_major_deity_dedication_level = { major_deity = vishnu level = 1 }
				}
				add_country_modifier = {
					name = re_religion_mechanic_hindu_modifier_vishnu_buddha_1
					duration = -1
					hidden = yes
				}
			}
			else = {
				add_country_modifier = {
					name = re_religion_mechanic_hindu_modifier_vishnu_buddha_0
					duration = -1
					hidden = yes
				}
			}
		}
	}
	hidden_effect = {
		if = {
			limit = {
				re_is_religion_mechanic_hindu_dedication_focus_enabled = yes
			}
			re_update_religion_mechanic_hindu_dynamic_dedication_focus_modifier = yes
		}
		if = {
			limit = {
				re_has_religion_mechanic_hindu_current_major_deity = { name = shiva }
			}
			set_variable = { which = re_religion_mechanic_hindu_current_value which = re_religion_mechanic_hindu_shiva_value }
		}
		else_if = {
			limit = {
				re_has_religion_mechanic_hindu_current_major_deity = { name = brahma }
			}
			set_variable = { which = re_religion_mechanic_hindu_current_value which = re_religion_mechanic_hindu_brahma_value }
		}
		else_if = {
			limit = {
				re_has_religion_mechanic_hindu_current_major_deity = { name = vishnu }
			}
			set_variable = { which = re_religion_mechanic_hindu_current_value which = re_religion_mechanic_hindu_vishnu_value }
		}
		else = {
			set_variable = { which = re_religion_mechanic_hindu_current_value value = 0 }
		}
	}
}
