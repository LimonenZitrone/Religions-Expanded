re_set_religion_mechanic_manichaeism_purity_value = {
	if = {
		limit = {
			re_uses_religion_mechanic = { mechanic = manichaeism_purity }
		}
		[[value]
			set_variable = { which = re_religion_mechanic_manichaeism_purity_value value = $value$ }
		]
		[[which]
			set_variable = { which = re_religion_mechanic_manichaeism_purity_value which = $which$ }
		]
		hidden_effect = {
			if = {
				limit = {
					re_has_religion_mechanic_manichaeism_purity_value = { value = 100 }
				}
				set_variable = { which = re_religion_mechanic_manichaeism_purity_value value = 100 }
			}
			else_if = {
				limit = {
					NOT = { re_has_religion_mechanic_manichaeism_purity_value = { value = 0 } }
				}
				set_variable = { which = re_religion_mechanic_manichaeism_purity_value value = 0 }
			}
			re_remove_religion_mechanic_manichaeism_purity_minor_focuses_if_too_many = yes
		}
	}
}

re_set_religion_mechanic_manichaeism_purity_liturgical_focus = {
	custom_tooltip = re_set_religion_mechanic_manichaeism_purity_liturgical_focus_$focus$_tt
	set_ruler_flag = re_religion_mechanic_manichaeism_purity_liturgical_focus_$focus$
}

re_add_religion_mechanic_manichaeism_purity_minor_focus = {
	if = {
		limit = {
			re_has_religion_mechanic_manichaeism_purity_liturgical_focus = { focus = moon }
		}
		set_ruler_flag = re_religion_mechanic_manichaeism_purity_minor_focus_$focus$
		add_ruler_modifier = {
			name = re_religion_mechanic_manichaeism_purity_$focus$_moon_modifier
			duration = -1
		}
	}
	else_if = {
		limit = {
			re_has_religion_mechanic_manichaeism_purity_liturgical_focus = { focus = sun }
		}
		set_ruler_flag = re_religion_mechanic_manichaeism_purity_minor_focus_$focus$
		add_ruler_modifier = {
			name = re_religion_mechanic_manichaeism_purity_$focus$_sun_modifier
			duration = -1
		}
	}
}

re_remove_religion_mechanic_manichaeism_purity_minor_focus = {
	clr_ruler_flag = re_religion_mechanic_manichaeism_purity_minor_focus_$focus$
	remove_country_modifier = re_religion_mechanic_manichaeism_purity_$focus$_moon_modifier
	remove_country_modifier = re_religion_mechanic_manichaeism_purity_$focus$_sun_modifier
}

re_export_monthly_religion_mechanic_manichaeism_purity_value_gain = {
	hidden_effect = {
		export_to_variable = {
			which = re_monthly_religion_mechanic_manichaeism_purity_value
			value = modifier:re_monthly_religion_mechanic_manichaeism_purity_change
		}
	}
}

re_add_monthly_religion_mechanic_manichaeism_purity_value = {
	re_export_monthly_religion_mechanic_manichaeism_purity_value_gain = yes
	re_add_religion_mechanic_manichaeism_purity_value = { which = re_monthly_religion_mechanic_manichaeism_purity_value }
}
