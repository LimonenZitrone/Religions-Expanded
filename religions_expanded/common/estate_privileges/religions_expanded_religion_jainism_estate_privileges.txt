re_religion_jainism_estate_jains_guaranteed_autonomy = {
	icon = privilege_grant_autonomy
	max_absolutism = -5
	conditional_modifier = { trigger = { has_absolutism_reduction_for_estate_privileges = yes } modifier = { max_absolutism = 1 } }
	loyalty = 0.05
	is_valid = {
		religion_group = dharmic
		NOT = { religion = jainism }
		any_owned_province = {
			religion = jainism
		}
		if = {
			limit = {
				is_expanded_mod_active = { mod = subjects }
			}
			NOT = { is_subject_of_type = iqta_vassal }
			NOT = { is_subject_of_type = religious_march }
		}
	}
	on_granted = {
		custom_tooltip = re_religion_jainism_estate_jains_guaranteed_autonomy_effect_tooltip
	}
	on_granted_province = {
		if = {
			limit = {
				religion = jainism
			}
			add_province_modifier = {
				name = re_religion_jainism_estate_jains_guaranteed_autonomy_mod
				duration = -1
			}
		}
		else_if = {
			limit = {
				has_province_modifier = re_religion_jainism_estate_jains_guaranteed_autonomy_mod
			}
			remove_province_modifier = re_religion_jainism_estate_jains_guaranteed_autonomy_mod
		}
	}
	on_revoked = {
		custom_tooltip = re_religion_jainism_estate_jains_guaranteed_autonomy_revoke_effect_tooltip
	}
	on_revoked_province = {
		if = {
			limit = {
				has_province_modifier = re_religion_jainism_estate_jains_guaranteed_autonomy_mod
			}
			remove_province_modifier = re_religion_jainism_estate_jains_guaranteed_autonomy_mod
		}
	}
	on_invalid = {
		custom_tooltip = re_religion_jainism_estate_jains_guaranteed_autonomy_revoke_effect_tooltip
	}
	on_invalid_province = {
		if = {
			limit = {
				has_province_modifier = re_religion_jainism_estate_jains_guaranteed_autonomy_mod
			}
			remove_province_modifier = re_religion_jainism_estate_jains_guaranteed_autonomy_mod
		}
	}
	penalties = {
	}
	benefits = {
	}
	ai_will_do = {
		factor = 1000
		modifier = {
			factor = 0
			NOT = {
				any_owned_province = {
					religion = jainism
				}
			}
		}
		modifier = {
			factor = 0
			is_subject = yes
			overlord = {
				religion = ROOT
				has_idea_group = religious_ideas
			}
		}
	}
}
