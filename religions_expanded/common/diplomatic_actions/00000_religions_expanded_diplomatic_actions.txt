		# Vanilla:
		change_asha_vahishta = { value = 1 }

		# Religions Expanded:
		re_add_religion_mechanic_sukha_value = { value = 3 }
		if = {
			limit = {

				NOT = { has_country_modifier = re_religion_mechanic_manichaeism_purity_gift_cooldown_modifier }
			}
			re_add_religion_mechanic_manichaeism_purity_value = { value = 3 }
			add_country_modifier = {
				name = re_religion_mechanic_manichaeism_purity_gift_cooldown_modifier
				duration = 3650
				hidden = yes
			}
		}
	}
}
