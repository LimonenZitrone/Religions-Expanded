on_startup = {
	if = {
		limit = {
			NOT = { has_global_flag = religions_expanded_religions_initialized }
			started_in = 1444.11.11
		}
		set_global_flag = religions_expanded_religions_initialized
		# coptic
		every_province = {
			limit = {
				re_religion_initialization_coptic_is_starting_province = yes
				NOT = { religion = coptic }
			}
			change_religion = coptic
		}
		# jainism
		every_province = {
			limit = {
				re_religion_initialization_jainism_is_starting_province = yes
				NOT = { religion = jainism }
			}
			change_religion = jainism
		}
		# tibetan_buddhism
		every_province = {
			limit = {
				re_religion_initialization_tibetan_buddhism_is_starting_province = yes
				NOT = { religion = tibetan_buddhism }
			}
			change_religion = tibetan_buddhism
		}
		every_country = {
			limit = {
				re_religion_initialization_tibetan_buddhism_is_starting_country = yes
				NOT = { religion = tibetan_buddhism }
			}
			set_country_flag = religions_expanded_initial_conversion
			change_religion = tibetan_buddhism
			clr_country_flag = religions_expanded_initial_conversion
		}
		# vidilist
		every_province = {
			limit = {
				re_religion_initialization_vidilist_is_starting_province = yes
				NOT = { religion = vidilist }
			}
			change_religion = vidilist
			add_permanent_province_modifier = {
				name = religious_zeal_at_conv
				duration = 1825
			}
		}
		# nestorian
		every_province = {
			limit = {
				re_religion_initialization_nestorian_is_starting_province = yes
				NOT = { religion = nestorian }
			}
			change_religion = nestorian
			add_permanent_province_modifier = {
				name = religious_zeal_at_conv
				duration = 1825
			}
		}
		# manichaeism
		every_province = {
			limit = {
				re_religion_initialization_manichaeism_is_starting_province = yes
				NOT = { religion = manichaeism }
			}
			change_religion = manichaeism
			add_permanent_province_modifier = {
				name = religious_zeal_at_conv
				duration = 1825
			}
		}
		# bogomilist
		every_province = {
			limit = {
				re_religion_initialization_bogomilist_is_starting_province = yes
				NOT = { religion = bogomilist }
			}
			change_religion = bogomilist
		}
	}
}
