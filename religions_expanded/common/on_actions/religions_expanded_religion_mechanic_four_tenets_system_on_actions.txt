on_religion_change = {
	re_clear_all_religion_mechanic_four_tenets_system_modifiers = yes
}

on_monthly_pulse = {
	if = {
		limit = {
			re_uses_religion_mechanic = { mechanic = four_tenets_system }
			NOT = { re_has_religion_mechanic_four_tenets_system_ability_4 = yes }
		}
		re_add_monthly_religion_mechanic_four_tenets_system_value = yes
	}
}

on_institution_embracement = {
	if = {
		limit = {
			re_uses_religion_mechanic = { mechanic = four_tenets_system }
		}
		if = {
			limit = {
				has_country_modifier = re_religion_mechanic_four_tenets_system_new_institution_modifier
			}
			remove_country_modifier = re_religion_mechanic_four_tenets_system_new_institution_modifier
		}
		add_country_modifier = {
			name = re_religion_mechanic_four_tenets_system_new_institution_modifier
			duration = 730
		}
		re_export_monthly_religion_mechanic_four_tenets_system_value_gain = yes
	}
}

on_tech_taken = {
	if = {
		limit = {
			re_uses_religion_mechanic = { mechanic = four_tenets_system }
		}
		if = {
			limit = {
				has_country_modifier = re_religion_mechanic_four_tenets_system_new_tech_modifier
			}

		}
		add_country_modifier = {
			name = re_religion_mechanic_four_tenets_system_new_tech_modifier
			duration = 730
		}
		re_export_monthly_religion_mechanic_four_tenets_system_value_gain = yes
	}
}

	FROM = {
		if = {
			limit = {
				re_uses_religion_mechanic = { mechanic = four_tenets_system }
			}
			re_clear_all_religion_mechanic_four_tenets_system_modifiers = yes
		}
	}
}
