on_religion_change = {
	rce_clear_all_religion_mechanic_sukha_modifiers_when_changing_religion = yes
}

on_monthly_pulse = {
	if = {
		limit = {
			rce_uses_religion_mechanic = { mechanic = sukha }
		}
		# yearly value / 12 is added
		rce_add_yearly_religion_mechanic_sukha_value = yes
	}
}

on_new_monarch = {
	if = {
		limit = {
			rce_uses_religion_mechanic = { mechanic = sukha }
		}
		set_variable = { which = rce_religion_mechanic_sukha_value_temp which = rce_religion_mechanic_sukha_value }
		divide_variable = { which = rce_religion_mechanic_sukha_value_temp value = 2 }
		rce_set_religion_mechanic_sukha_value = { which = rce_religion_mechanic_sukha_value_temp }
		set_variable = { which = rce_religion_mechanic_sukha_value_temp value = 0 }
	}
}

# ROOT = Province, FROM = Country who Propogated
on_convert_by_trade_policy = {
	FROM = {
		if = {
			limit = {
				rce_uses_religion_mechanic = { mechanic = sukha }
			}
			export_to_variable = {
				which = rce_religion_mechanic_sukha_prov_dev
				value = trigger_value:development
				who = ROOT
			}
			multiply_variable = { which = rce_religion_mechanic_sukha_prov_dev value = 0.1 }
			rce_add_religion_mechanic_sukha_value = { which = rce_religion_mechanic_sukha_prov_dev }
			set_variable = { which = rce_religion_mechanic_sukha_prov_dev value = 0 }
		}
	}
}

			}
			rce_clear_all_religion_mechanic_sukha_modifiers = yes
		}

}

# ROOT = released country, FROM = releasing country
on_country_released = {
	if = {
		limit = {
			rce_uses_religion_mechanic = { mechanic = sukha }
		}
		rce_clear_all_religion_mechanic_sukha_modifiers = yes
		rce_set_religion_mechanic_sukha_value = { value = 0 }
	}
	else = {
		rce_clear_all_religion_mechanic_sukha_modifiers_when_changing_religion = yes
	}
}
