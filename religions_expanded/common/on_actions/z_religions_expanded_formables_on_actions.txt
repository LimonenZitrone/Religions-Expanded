on_annexed = {
	FROM = {
		if = {
			limit = {
				has_country_flag = re_formable_assyria
			}
			clr_global_flag = re_formable_assyria
			clr_country_flag = re_formable_assyria
			restore_country_name = yes
			restore_country_color = yes
			swap_free_idea_group = yes
		}
		if = {
			limit = {
				has_country_flag = re_formable_empire_of_light
			}
			clr_global_flag = re_formable_empire_of_light
			clr_country_flag = re_formable_empire_of_light
			restore_country_name = yes
			restore_country_color = yes
			swap_free_idea_group = yes
		}
	}
}
