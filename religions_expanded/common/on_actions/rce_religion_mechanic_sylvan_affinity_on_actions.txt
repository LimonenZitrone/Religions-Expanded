on_religion_change = {
	rce_clear_all_religion_mechanic_sylvan_affinity_modifiers = yes
}

on_monthly_pulse = {
	if = {
		limit = {
			rce_uses_religion_mechanic = { mechanic = sylvan_affinity }
		}
		rce_add_monthly_religion_mechanic_sylvan_affinity_value = yes
	}
}

# ROOT = annexing country, FROM = annexed country
on_annexed = {
	FROM = {
		if = {
			limit = {
				rce_uses_religion_mechanic = { mechanic = sylvan_affinity }
			}
			rce_clear_all_religion_mechanic_sylvan_affinity_modifiers = yes
		}
	}
}

# ROOT = released country, FROM = releasing country
on_country_released = {
	rce_clear_all_religion_mechanic_sylvan_affinity_modifiers = yes
}

# ROOT = released country
on_country_creation = {
	rce_clear_all_religion_mechanic_sylvan_affinity_modifiers = yes
}

# ROOT = vassal, FROM = overlord
on_create_vassal = {
	rce_clear_all_religion_mechanic_sylvan_affinity_modifiers = yes
}
