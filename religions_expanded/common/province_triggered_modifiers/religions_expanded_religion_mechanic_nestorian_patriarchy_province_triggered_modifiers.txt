re_religion_mechanic_nestorian_patriarchy_center_of_reformation_province_triggered_modifier = {
	hidden = yes

	potential = {
		always = yes
	}

	trigger = {
		OR = {
			NOT = { is_month = 1 }
			AND = {
				is_month = 2
				NOT = { is_month = 3 }
			}
			AND = {
				is_month = 4
				NOT = { is_month = 5 }
			}
			AND = {
				is_month = 6
				NOT = { is_month = 7 }
			}
			AND = {
				is_month = 8
				NOT = { is_month = 9 }
			}
			AND = {
				is_month = 10
				NOT = { is_month = 11 }
			}
		}
	}

	on_activation = {
		if = {
			limit = {
				NOT = { has_province_flag = re_religion_mechanic_nestorian_patriarchy_center_of_reformation_setup }
			}
			if = {
				limit = {
					NOT = { has_saved_global_event_target = re_religion_mechanic_nestorian_patriarchy_current_center_of_reformation_target }
				}
				re_religion_mechanic_nestorian_patriarchy_pick_center_of_reformation_target = yes
			}
			else = {
				if = {
					limit = {
						check_variable = { which = re_religion_mechanic_nestorian_patriarchy_center_of_reformation_conversion_time value = 18 }
					}
					event_target:re_religion_mechanic_nestorian_patriarchy_current_center_of_reformation_target = {
						province_event = { id = re_religion_mechanic_nestorian_patriarchy_events.2 }
					}
					set_variable = { which = re_religion_mechanic_nestorian_patriarchy_center_of_reformation_conversion_time value = 0 }
					clear_global_event_target = re_religion_mechanic_nestorian_patriarchy_current_center_of_reformation_target
				}
				else = {
					change_variable = { which = re_religion_mechanic_nestorian_patriarchy_center_of_reformation_conversion_time value = 1 }
				}
			}
		}
		else = {
			clr_province_flag = re_religion_mechanic_nestorian_patriarchy_center_of_reformation_setup
		}
	}

	on_deactivation = {
		if = {
			limit = {
				NOT = { has_province_flag = re_religion_mechanic_nestorian_patriarchy_center_of_reformation_setup }
			}
			if = {
				limit = {
					NOT = { has_saved_global_event_target = re_religion_mechanic_nestorian_patriarchy_current_center_of_reformation_target }
				}
				re_religion_mechanic_nestorian_patriarchy_pick_center_of_reformation_target = yes
			}
			else = {
				if = {
					limit = {
						check_variable = { which = re_religion_mechanic_nestorian_patriarchy_center_of_reformation_conversion_time value = 18 }
					}
					event_target:re_religion_mechanic_nestorian_patriarchy_current_center_of_reformation_target = {
						province_event = { id = re_religion_mechanic_nestorian_patriarchy_events.2 }
					}
					set_variable = { which = re_religion_mechanic_nestorian_patriarchy_center_of_reformation_conversion_time value = 0 }
					clear_global_event_target = re_religion_mechanic_nestorian_patriarchy_current_center_of_reformation_target
				}
				else = {
					change_variable = { which = re_religion_mechanic_nestorian_patriarchy_center_of_reformation_conversion_time value = 1 }
				}
			}
		}
		else = {
			clr_province_flag = re_religion_mechanic_nestorian_patriarchy_center_of_reformation_setup
		}
	}
}
