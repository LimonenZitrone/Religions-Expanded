rce_uses_religion_mechanic_hindu = {
	custom_trigger_tooltip = {
		tooltip = rce_uses_religion_mechanic_hindu_tt
		religion = hinduism
		uses_personal_deities = yes
	}
}

rce_not_uses_religion_mechanic_hindu = {
	custom_trigger_tooltip = {
		tooltip = rce_not_uses_religion_mechanic_hindu_tt
		NOT = { rce_uses_religion_mechanic_hindu = yes }
	}
}

rce_has_religion_mechanic_hindu_major_deity_current_dedication_level = {
	if = {
		limit = {
			rce_uses_religion_mechanic = { mechanic = hindu }
		}
		if = {
			limit = {
				rce_has_religion_mechanic_hindu_current_major_deity = { name = shiva }
			}
			rce_has_religion_mechanic_hindu_major_deity_shiva_dedication_level = { level = $level$ }
		}
		else_if = {
			limit = {
				rce_has_religion_mechanic_hindu_current_major_deity = { name = brahma }
			}
			rce_has_religion_mechanic_hindu_major_deity_brahma_dedication_level = { level = $level$ }
		}
		else_if = {
			limit = {
				rce_has_religion_mechanic_hindu_current_major_deity = { name = vishnu }
			}
			rce_has_religion_mechanic_hindu_major_deity_vishnu_dedication_level = { level = $level$ }
		}
		else = {
			hidden_trigger = { always = no }
		}
	}
}
