rce_uses_religion_mechanic_four_tenets_system = {
	custom_trigger_tooltip = {
		tooltip = rce_uses_religion_mechanic_four_tenets_system_tt
		religion = tibetan_buddhism
	}
}

rce_not_uses_religion_mechanic_four_tenets_system = {
	custom_trigger_tooltip = {
		tooltip = rce_not_uses_religion_mechanic_four_tenets_system_tt
		NOT = { rce_uses_religion_mechanic_four_tenets_system = yes }
	}
}

rce_has_religion_mechanic_four_tenets_system_value = {
	if = {
		limit = {
			rce_uses_religion_mechanic = { mechanic = four_tenets_system }
		}
		[[value]
			check_variable = { which = rce_religion_mechanic_four_tenets_system_value value = $value$ }
		]
		[[which]
			check_variable = { which = rce_religion_mechanic_four_tenets_system_value which = $which$ }
		]
	}
}
