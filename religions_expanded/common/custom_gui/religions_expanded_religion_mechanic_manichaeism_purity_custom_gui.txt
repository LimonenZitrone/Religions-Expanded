custom_button = {
	name = re_religion_mechanic_manichaeism_purity_wind_button
	# if you change this, make sure to change the re_religion_mechanic_manichaeism_purity_wind_decision decision accordingly
	trigger = {
		if = {
			limit = {
				NOT = { re_has_religion_mechanic_manichaeism_purity_minor_focus = { focus = wind } }
			}
			re_has_any_religion_mechanic_manichaeism_purity_liturgical_focus = yes
			re_religion_mechanic_manichaeism_purity_has_fewer_than_allowed_minor_focuses = yes
		}
	}
	effect = {
		custom_tooltip = religions_expanded_tooltip_effect_text
		if = {
			limit = {
				NOT = { re_has_religion_mechanic_manichaeism_purity_minor_focus = { focus = wind } }
			}
			hidden_effect = {
				re_add_religion_mechanic_manichaeism_purity_minor_focus = { focus = wind }
			}
		}
		else = {
			re_remove_religion_mechanic_manichaeism_purity_minor_focus = { focus = wind }
			custom_tooltip = re_religion_mechanic_manichaeism_purity_focus_button_tt_currently_active
		}
		if = {
			limit = {
				re_has_religion_mechanic_manichaeism_purity_liturgical_focus = { focus = moon }
			}
			custom_tooltip = re_religion_mechanic_manichaeism_purity_focus_button_tt_moon_yes
		}
		else = {
			custom_tooltip = re_religion_mechanic_manichaeism_purity_focus_button_tt_moon_no
		}
		tooltip = {
			add_country_modifier = {
				name = re_religion_mechanic_manichaeism_purity_wind_moon_modifier
				duration = -1
				desc = re_religion_mechanic_manichaeism_purity_until_ruler_death_or_focus_disabled_tt
			}
		}
		if = {
			limit = {
				re_has_religion_mechanic_manichaeism_purity_liturgical_focus = { focus = sun }
			}
			custom_tooltip = re_religion_mechanic_manichaeism_purity_focus_button_tt_sun_yes
		}
		else = {
			custom_tooltip = re_religion_mechanic_manichaeism_purity_focus_button_tt_sun_no
		}
		tooltip = {
			add_country_modifier = {
				name = re_religion_mechanic_manichaeism_purity_wind_sun_modifier
				duration = -1
				desc = re_religion_mechanic_manichaeism_purity_until_ruler_death_or_focus_disabled_tt
			}
		}
	}
	frame = {
		number = 2
		trigger = {
			re_has_religion_mechanic_manichaeism_purity_minor_focus = { focus = wind }
		}
