custom_button = {
	name = rce_religion_mechanic_sukha_meditate_button
	# if you change this, make sure to change the rce_religion_mechanic_sukha_meditate_decision decision accordingly
	trigger = {
		adm_power = 50
		NOT = { rce_has_religion_mechanic_sukha_value = { value = 100 } }
		if = {
			limit = {
				has_country_flag = rce_religion_mechanic_sukha_meditate_used
			}
			custom_trigger_tooltip = {
				tooltip = rce_religion_mechanic_sukha_meditate_not_used_recently_tt_with_year
				had_country_flag = { flag = rce_religion_mechanic_sukha_meditate_used days = 3650 }
			}
		}
		else = {
			custom_trigger_tooltip = {
				tooltip = rce_religion_mechanic_sukha_meditate_not_used_recently_tt
				always = yes
			}
		}
	}
	effect = {
		custom_tooltip = rce_tooltip_effect_text
		add_adm_power = -50
		rce_add_religion_mechanic_sukha_value = { value = 10 }
		hidden_effect = {
			clr_country_flag = rce_religion_mechanic_sukha_meditate_used
			set_country_flag = rce_religion_mechanic_sukha_meditate_used
			export_to_variable = {
				which = rce_religion_mechanic_sukha_meditate_used_year
				value = trigger_value:is_year
			}
		}
	}
	tooltip = rce_religion_mechanic_sukha_meditate_button_tt
}
custom_button = {
	name = rce_religion_mechanic_sukha_sponsor_festival_button
	# if you change this, make sure to change the rce_religion_mechanic_sukha_sponsor_festival_decision decision accordingly
	trigger = {
		if = {
			limit = {
				monthly_income = 500
			}
			treasury = 3000
		}
		else_if = {
			limit = {
				NOT = { monthly_income = 17 }
			}
			treasury = 100
		}
		else = {
			years_of_income = 0.5
		}
		if = {
			limit = {
				has_country_flag = rce_religion_mechanic_sukha_sponsor_festival_used
			}
			custom_trigger_tooltip = {
				tooltip = rce_religion_mechanic_sukha_sponsor_festival_not_used_recently_tt_with_year

		}
		else = {
			custom_trigger_tooltip = {
				tooltip = rce_religion_mechanic_sukha_sponsor_festival_not_used_recently_tt
				always = yes
			}
		}
	}
	effect = {
		custom_tooltip = rce_tooltip_effect_text
		if = {
			limit = {

			}
			add_treasury = -3000
		}

			}

		}
		else = {
			add_years_of_income = -0.5
		}
		custom_tooltip = rce_religion_mechanic_sukha_sponsor_festival_button_tt_choice

			add_country_modifier = {
				name = rce_religion_mechanic_sukha_sponsor_festival_paryushana_modifier

			}
		}
		hidden_effect = {
			country_event = { id = rce_religion_mechanic_sukha_events.1 }
			clr_country_flag = rce_religion_mechanic_sukha_sponsor_festival_used
