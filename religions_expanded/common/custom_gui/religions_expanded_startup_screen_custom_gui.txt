# Main Window
custom_window = {
	name = religions_expanded_startup_screen_window
	potential = {
		ai = no
		has_country_flag = show_religions_expanded_startup_screen_window
	}
}
custom_button = {
	name = religions_expanded_startup_screen_close_button
	effect = {
		hidden_effect = {
			clr_country_flag = show_religions_expanded_startup_screen_window
			clr_country_flag = show_religions_expanded_startup_screen_window_religions
			clr_country_flag = show_religions_expanded_startup_screen_window_cultures
			set_variable = { which = religions_expanded_startup_screen_window_selected_religion value = 0 }
			set_variable = { which = religions_expanded_startup_screen_window_selected_culture value = 0 }
		}
	}
}
