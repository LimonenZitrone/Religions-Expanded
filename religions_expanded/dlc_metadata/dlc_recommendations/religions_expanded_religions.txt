# DLC info for the lobby used to show information for new religions and mechanics
# Country Info
religions_expanded_entry_country_religion_info_hinduism = {
	reason = {
		tooltip_desc = "religions_expanded_lobby_info_entry_tooltip"
		tooltip_reason = ""
		weight = 60
		trigger = {
			NOT = { has_global_flag = religions_expanded_religions_initialized }
			religion = hinduism
		}
	}
}
# Country
religions_expanded_entry_country_change_religion_tibetan_buddhism = {
	reason = {
		tooltip_desc = "religions_expanded_lobby_info_religion_entry_tooltip"

		weight = 50
		trigger = {
			NOT = { has_global_flag = religions_expanded_religions_initialized }
			re_religion_initialization_tibetan_buddhism_is_starting_country = yes

}
