namespace = rce_culture_byzantine_events

province_event = {
	id = rce_culture_byzantine_events.1
	title = rce_culture_byzantine_events.EVTNAME1
	desc = rce_culture_byzantine_events.EVTDESC1
	picture = CULTURE_eventPicture

	trigger = {
		OR = {
			area = kastamonu_area
			province_id = 330
		}
		has_owner_religion = yes
		NOT = { culture = pontic_greek }
		NOT = { has_province_flag = rce_converted_to_pontic_greek }
		owner = {
			culture_group = byzantine
			OR = {
				religion_group = christian
				religion = hellenism
			}
			OR = {
				government_rank = 3
				AND = {
					NOT = { has_dlc = "Common Sense" }
					total_development = 1000
				}
			}
		}
	}

	mean_time_to_happen = {
		months = 600
		modifier = {
			factor = 0.5
			owner = {
				religion = hellenism
			}
		}
		modifier = {
			factor = 0.8
			any_neighbor_province = {
				owned_by = ROOT
				culture_group = byzantine
			}
		}
		modifier = {
			factor = 0.7
			calc_true_if = {
				amount = 3
				all_neighbor_province = {
					owned_by = ROOT
					culture_group = byzantine
				}
			}
		}
	}

	immediate = {
		hidden_effect = {
			set_province_flag = rce_converted_to_pontic_greek
		}
	}

	option = {
		name = rce_culture_byzantine_events1.EVTOPT1

		change_culture = pontic_greek
	}
}
