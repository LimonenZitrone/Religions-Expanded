on_annexed = {
	FROM = {
		if = {
			limit = {
				has_country_flag = rce_formable_assyria
			}
			clr_global_flag = rce_formable_assyria
			clr_country_flag = rce_formable_assyria
			restore_country_name = yes
			restore_country_color = yes
			swap_free_idea_group = yes
		}
		if = {
			limit = {
				has_country_flag = rce_formable_empire_of_light
			}
			clr_global_flag = rce_formable_empire_of_light
			clr_country_flag = rce_formable_empire_of_light
			restore_country_name = yes
			restore_country_color = yes
			swap_free_idea_group = yes
		}
		if = {
			limit = {
				has_country_flag = rce_formable_berber_mauretania
			}
			clr_global_flag = rce_formable_berber_mauretania
			clr_country_flag = rce_formable_berber_mauretania
			restore_country_name = yes
			restore_country_color = yes
			swap_free_idea_group = yes
		}
	}
}

on_primary_culture_changed = {
	if = {
		limit = {
			has_country_flag = rce_formable_assyria
		}
		clr_global_flag = rce_formable_assyria
		clr_country_flag = rce_formable_assyria
		restore_country_name = yes
		restore_country_color = yes
		swap_free_idea_group = yes
	}
	if = {
		limit = {
			has_country_flag = rce_formable_berber_mauretania
			NOT = { culture_group = maghrebi }
		}
		clr_global_flag = rce_formable_berber_mauretania
		clr_country_flag = rce_formable_berber_mauretania
		restore_country_name = yes
		restore_country_color = yes
		swap_free_idea_group = yes
	}
}

on_religion_change = {
	if = {
		limit = {
			has_country_flag = rce_formable_empire_of_light
		}
		clr_global_flag = rce_formable_empire_of_light
		clr_country_flag = rce_formable_empire_of_light
		restore_country_name = yes
		restore_country_color = yes
		swap_free_idea_group = yes
	}
}

on_province_owner_change = {
	if = {
		limit = {
			region = maghreb_region
			FROM = {
				has_country_flag = rce_formable_berber_mauretania
				NOT = {
					any_owned_province = {
						region = maghreb_region
						culture_group = maghrebi
					}
				}
			}
		}
		FROM = {
			clr_global_flag = rce_formable_berber_mauretania
			clr_country_flag = rce_formable_berber_mauretania
			restore_country_name = yes
			restore_country_color = yes
			swap_free_idea_group = yes
		}
	}
}
