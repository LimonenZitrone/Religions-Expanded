on_startup = {
	rce_give_religion_mechanic_norse_random_era_if_none_present = yes
	rce_set_religion_mechanic_pantheon_image_frame_for_non_selected_gods = yes
}

on_religion_change = {
	rce_clear_all_religion_mechanic_pantheon_modifiers = yes
	rce_set_religion_mechanic_pantheon_image_frame_for_non_selected_gods = yes
}

# ROOT = annexing country, FROM = annexed country
on_annexed = {
	FROM = {
		if = {
			limit = {
				rce_uses_religion_mechanic = { mechanic = pantheon }
			}
			rce_clear_all_religion_mechanic_pantheon_modifiers = yes
		}
	}
}

# ROOT = released country, FROM = releasing country
on_country_released = {
	rce_clear_all_religion_mechanic_pantheon_modifiers = yes
	rce_set_religion_mechanic_pantheon_image_frame_for_non_selected_gods = yes
}

# ROOT = released country
on_country_creation = {
	rce_clear_all_religion_mechanic_pantheon_modifiers = yes
	rce_set_religion_mechanic_pantheon_image_frame_for_non_selected_gods = yes
}

# ROOT = vassal, FROM = overlord
on_create_vassal = {
	rce_clear_all_religion_mechanic_pantheon_modifiers = yes
	rce_set_religion_mechanic_pantheon_image_frame_for_non_selected_gods = yes
}
