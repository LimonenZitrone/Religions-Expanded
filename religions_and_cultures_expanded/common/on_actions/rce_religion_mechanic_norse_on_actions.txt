on_startup = {
	rce_give_religion_mechanic_norse_random_era_if_none_present = yes
}

on_religion_change = {
	rce_clear_all_religion_mechanic_norse_modifiers = yes
	rce_give_religion_mechanic_norse_random_era_if_none_present = yes
}

on_monthly_pulse = {
	if = {
		limit = {
			rce_uses_religion_mechanic = { mechanic = norse }
		}
		rce_add_monthly_religion_mechanic_norse_values = yes
	}
}

	rce_add_religion_mechanic_norse_norn_value = {
