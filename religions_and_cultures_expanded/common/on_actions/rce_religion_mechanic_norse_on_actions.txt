on_religion_change = {
	rce_clear_all_religion_mechanic_norse_modifiers_when_changing_religion = yes
	if = {
		limit = {
			rce_uses_religion_mechanic = { mechanic = norse }
		}
		random_list = {
			33 = {
				modifier = {
					factor = 3
					ADM = 6
				}

					factor = 2
					ADM = 5
				}
				modifier = {
					factor = 0.9
					NOT = { ADM = 4 }
				}

					factor = 0.75
					NOT = { ADM = 3 }
				}
				modifier = {
					factor = 0.5
					NOT = { ADM = 2 }
				}

					factor = 0.25
					NOT = { ADM = 1 }
				}

on_monthly_pulse = {
	if = {
		limit = {
			rce_uses_religion_mechanic = { mechanic = norse }
		}
		rce_add_monthly_religion_mechanic_norse_values = yes
	}
}

# ROOT = annexing country, FROM = annexed country
on_annexed = {
	FROM = {
		if = {
			limit = {
				rce_uses_religion_mechanic = { mechanic = norse }
			}
			rce_clear_all_religion_mechanic_norse_modifiers = yes
		}
	}
}

# ROOT = released country, FROM = releasing country
on_country_released = {
	if = {
		limit = {
			rce_uses_religion_mechanic = { mechanic = norse }
		}
		rce_clear_all_religion_mechanic_norse_modifiers = yes
	}
	else = {
		rce_clear_all_religion_mechanic_norse_modifiers_when_changing_religion = yes
	}
}
