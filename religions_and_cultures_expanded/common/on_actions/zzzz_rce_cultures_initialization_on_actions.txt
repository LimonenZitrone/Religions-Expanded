on_startup = {
	if = {
		limit = {
			NOT = { has_global_flag = rce_cultures_initialized }
			started_in = 1444.11.11
		}
		set_global_flag = rce_cultures_initialized
		# coptic_culture
		every_province = {
			limit = {
				rce_culture_initialization_coptic_culture_is_starting_province = yes
				NOT = { culture = coptic_culture }
			}
			change_culture = coptic_culture
		}
		# assyrian_culture
		every_province = {
			limit = {
				rce_culture_initialization_assyrian_culture_is_starting_province = yes
				NOT = { culture = assyrian_culture }
			}
			change_culture = assyrian_culture
		}
		# avshars_culture
		every_province = {
			limit = {
				rce_culture_initialization_avshars_culture_is_starting_province = yes
				NOT = { culture = avshars_culture }
			}
			change_culture = avshars_culture
		}
		# qashqai_culture
		every_province = {
			limit = {
				rce_culture_initialization_qashqai_culture_is_starting_province = yes
				NOT = { culture = qashqai_culture }
			}
			change_culture = qashqai_culture
		}
		# azeri_culture
		every_province = {
			limit = {
				rce_culture_initialization_azeri_culture_is_starting_province = yes
				NOT = { culture = azeri_culture }
			}
			change_culture = azeri_culture
		}
		every_country_including_inactive = {
			limit = {
				rce_culture_initialization_azeri_culture_is_starting_country = yes
				NOT = { primary_culture = azeri_culture }
			}
			change_primary_culture = azeri_culture
			if = {
				limit = {
					exists = yes
				}
				if = {
					limit = {
						ruler_culture = azerbaijani
					}
					set_ruler_culture = azeri_culture
				}
				if = {
					limit = {
						has_heir = yes
						heir_culture = azerbaijani
					}
					set_heir_culture = azeri_culture
				}
				if = {
					limit = {
						has_consort = yes
						consort_culture = azerbaijani
					}
					set_consort_culture = azeri_culture
				}
			}
		}
		# oghuz_culture
		every_province = {
			limit = {
				rce_culture_initialization_oghuz_culture_is_starting_province = yes
				NOT = { culture = oghuz_culture }
			}
			change_culture = oghuz_culture
		}
		every_country_including_inactive = {
			limit = {
				rce_culture_initialization_oghuz_culture_is_starting_country = yes
				NOT = { primary_culture = oghuz_culture }
			}
			change_primary_culture = oghuz_culture
			if = {
				limit = {
					exists = yes
				}
				if = {
					limit = {
						ruler_culture = turkish
					}
					set_ruler_culture = oghuz_culture
				}
				if = {
					limit = {
						has_heir = yes
						heir_culture = turkish
					}
					set_heir_culture = oghuz_culture
				}
				if = {
					limit = {
						has_consort = yes
						consort_culture = turkish
					}
					set_consort_culture = oghuz_culture
				}
			}
		}
		# farsi_culture
		every_province = {
			limit = {
				rce_culture_initialization_farsi_culture_is_starting_province = yes
				NOT = { culture = farsi_culture }
			}
			change_culture = farsi_culture
		}
		every_country_including_inactive = {
			limit = {
				rce_culture_initialization_farsi_culture_is_starting_country = yes
				NOT = { primary_culture = farsi_culture }
			}
			change_primary_culture = farsi_culture
			if = {
				limit = {
					exists = yes
				}
				if = {
					limit = {
						ruler_culture = persian
					}
					set_ruler_culture = farsi_culture
				}
				if = {
					limit = {
						has_heir = yes
						heir_culture = persian
					}
					set_heir_culture = farsi_culture
				}
				if = {
					limit = {
						has_consort = yes
						consort_culture = persian
					}
					set_consort_culture = farsi_culture
				}
			}
		}
		# manichaean_culture
		every_province = {
			limit = {
				rce_culture_initialization_manichaean_culture_is_starting_province = yes
				NOT = { culture = manichaean_culture }
			}
			change_culture = manichaean_culture
		}
	}
}
