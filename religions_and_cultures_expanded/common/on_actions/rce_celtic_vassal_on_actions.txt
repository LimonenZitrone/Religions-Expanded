# ROOT = overlord, FROM = subject
on_dependency_gained = {
	if = {
		limit = {
			NOT = { can_be_overlord = rce_celtic_vassal }
			FROM = {
				is_subject_of_type = rce_celtic_vassal
			}
		}
		create_subject = {
			subject = FROM
			subject_type = vassal
			keep_liberty_desire = yes
		}
		FROM = {
			restore_country_name = yes
			restore_country_color = yes
		}
	}
}

on_primary_culture_changed = {
	if = {
		limit = {
			is_subject = yes
			is_subject_of_type = rce_celtic_vassal
		}
		restore_country_name = yes
		restore_country_color = yes
		overlord = {
			create_subject = {
				subject = ROOT
				subject_type = vassal
				keep_liberty_desire = yes
			}
		}
	}
	if = {
		limit = {
			NOT = { can_be_overlord = rce_celtic_vassal }
			has_subject_of_type = rce_celtic_vassal
		}
		every_subject_country = {
			limit = {
				is_subject_of_type = rce_celtic_vassal
			}
			restore_country_name = yes
			restore_country_color = yes
			ROOT = {
				create_subject = {
					subject = PREV
					subject_type = vassal
					keep_liberty_desire = yes
				}
			}
		}
	}
}

on_monthly_pulse = {
	if = {
		limit = {
			can_be_overlord = rce_celtic_vassal
			has_subject_of_type = rce_celtic_vassal
		}
		every_subject_country = {
			limit = {
				is_subject_of_type = rce_celtic_vassal
			}
			change_country_color = { country = ROOT }
		}
	}
}
