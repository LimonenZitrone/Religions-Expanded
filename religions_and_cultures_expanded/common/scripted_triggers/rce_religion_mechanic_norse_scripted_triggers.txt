rce_uses_religion_mechanic_norse = {
	custom_trigger_tooltip = {
		tooltip = rce_uses_religion_mechanic_norse_tt
		religion = norse_pagan_reformed
		uses_personal_deities = yes
	}
}

rce_not_uses_religion_mechanic_norse = {
	custom_trigger_tooltip = {
		tooltip = rce_not_uses_religion_mechanic_norse_tt
		NOT = { rce_uses_religion_mechanic_norse = yes }
	}
}

rce_has_religion_mechanic_norse_era = {
	custom_trigger_tooltip = {
		tooltip = rce_has_religion_mechanic_norse_era_tt_$era$
		has_country_flag = religion_mechanic_norse_era_$era$
	}
}

rce_has_not_religion_mechanic_norse_era = {
	custom_trigger_tooltip = {
		tooltip = rce_has_not_religion_mechanic_norse_era_tt_$era$
		NOT = { has_country_flag = religion_mechanic_norse_era_$era$ }
	}
}

rce_has_religion_mechanic_norse_norn_value = {
	if = {
		limit = {
			rce_uses_religion_mechanic = { mechanic = norse }
		}
		[[value]
			check_variable = { which = rce_religion_mechanic_norse_norn_$type$_value value = $value$ }
		]
		[[which]
			check_variable = { which = rce_religion_mechanic_norse_norn_$type$_value which = $which$ }
		]
	}
}

rce_has_religion_mechanic_norse_deity = {
	if = {
		limit = {
			rce_uses_religion_mechanic = { mechanic = norse }

rce_is_religion_mechanic_norse_next_era_adm = {
	custom_trigger_tooltip = {
		tooltip = rce_is_religion_mechanic_norse_next_era_adm_tt
		variable_arithmetic_trigger = {
			export_to_variable = {
				which = rce_religion_mechanic_norse_norn_adm_value_temp_1
				value = 0
			}
			export_to_variable = {
				which = rce_religion_mechanic_norse_norn_adm_value_temp_2
				value = 0
			}
			change_variable = { which = rce_religion_mechanic_norse_norn_adm_value_temp_1 which = rce_religion_mechanic_norse_norn_adm_value }
			change_variable = { which = rce_religion_mechanic_norse_norn_adm_value_temp_2 which = rce_religion_mechanic_norse_norn_adm_value }
			subtract_variable = { which = rce_religion_mechanic_norse_norn_adm_value_temp_1 which = rce_religion_mechanic_norse_norn_dip_value }
			subtract_variable = { which = rce_religion_mechanic_norse_norn_adm_value_temp_2 which = rce_religion_mechanic_norse_norn_mil_value }
			check_variable = { which = rce_religion_mechanic_norse_norn_adm_value which = rce_religion_mechanic_norse_norn_dip_value }
			check_variable = { which = rce_religion_mechanic_norse_norn_adm_value which = rce_religion_mechanic_norse_norn_mil_value }
			OR = {
				NOT = {
					check_variable = { which = rce_religion_mechanic_norse_norn_adm_value value = 10 }
					check_variable = { which = rce_religion_mechanic_norse_norn_dip_value value = 10 }
					check_variable = { which = rce_religion_mechanic_norse_norn_mil_value value = 10 }
				}
				AND = {
					OR = {
						NOT = { check_variable = { which = rce_religion_mechanic_norse_norn_adm_value_temp_1 value = -5 } }
						check_variable = { which = rce_religion_mechanic_norse_norn_adm_value_temp_1 value = 5.001 }
					}
					OR = {
						NOT = { check_variable = { which = rce_religion_mechanic_norse_norn_adm_value_temp_2 value = -5 } }
						check_variable = { which = rce_religion_mechanic_norse_norn_adm_value_temp_2 value = 5.001 }
					}
				}
			}
		}
	}
}
