rce_uses_religion_mechanic_sylvan_affinity = {
	custom_trigger_tooltip = {
		tooltip = rce_uses_religion_mechanic_sylvan_affinity_tt
		religion = vidilist
	}
}

rce_not_uses_religion_mechanic_sylvan_affinity = {
	custom_trigger_tooltip = {
		tooltip = rce_not_uses_religion_mechanic_sylvan_affinity_tt
		NOT = { rce_uses_religion_mechanic_sylvan_affinity = yes }
	}
}

rce_has_religion_mechanic_sylvan_affinity_value = {
	if = {
		limit = {
			rce_uses_religion_mechanic = { mechanic = sylvan_affinity }
		}
		[[value]
			check_variable = { which = rce_religion_mechanic_sylvan_affinity_value value = $value$ }
		]
		[[which]
			check_variable = { which = rce_religion_mechanic_sylvan_affinity_value which = $which$ }
		]
	}
}
