rce_update_religion_mechanic_hindu_dedication_levels = {
	hidden_effect = {
		rce_update_religion_mechanic_hindu_dedication_levels_specific = { major_deity = shiva }
		rce_update_religion_mechanic_hindu_dedication_levels_specific = { major_deity = brahma }
		rce_update_religion_mechanic_hindu_dedication_levels_specific = { major_deity = vishnu }
		rce_update_religion_mechanic_hindu_dynamic_modifier = yes
		rce_export_monthly_religion_mechanic_hindu_value_gain = yes
	}
}
rce_update_religion_mechanic_hindu_dedication_levels_specific = {
	while = {
		limit = {
			check_variable = { which = rce_religion_mechanic_hindu_$major_deity$_value value = 100 }
			OR = {
				NOT = { is_variable_equal = { which = rce_religion_mechanic_hindu_$major_deity$_value value = 100 } }
				NOT = { rce_has_religion_mechanic_hindu_major_deity_dedication_level = { major_deity = $major_deity$ level = 5 } }
			}
		}
		if = {
			limit = {
				rce_has_religion_mechanic_hindu_major_deity_dedication_level = { major_deity = $major_deity$ level = 5 }
			}
			set_variable = { which = rce_religion_mechanic_hindu_$major_deity$_value value = 100 }
		}
		else = {
			subtract_variable = { which = rce_religion_mechanic_hindu_$major_deity$_value value = 100 }
			change_variable = { which = rce_religion_mechanic_hindu_$major_deity$_level value = 1 }
		}
	}
	while = {
		limit = {
			NOT = { check_variable = { which = rce_religion_mechanic_hindu_$major_deity$_value value = 0 } }
		}
		if = {
			limit = {
				NOT = { rce_has_religion_mechanic_hindu_major_deity_dedication_level = { major_deity = $major_deity$ level = 1 } }
			}
			set_variable = { which = rce_religion_mechanic_hindu_$major_deity$_value value = 0 }
		}
		else = {
			change_variable = { which = rce_religion_mechanic_hindu_$major_deity$_value value = 100 }
			subtract_variable = { which = rce_religion_mechanic_hindu_$major_deity$_level value = 1 }
		}
	}
}

rce_export_monthly_religion_mechanic_hindu_value_gain = {
	hidden_effect = {
		if = {
			limit = {
				OR = {
					rce_has_religion_mechanic_hindu_current_major_deity = { name = shiva }

					rce_has_religion_mechanic_hindu_current_major_deity = { name = vishnu }
				}
			}
			set_variable = { which = rce_monthly_religion_mechanic_hindu_value value = 0 }
			export_to_variable = {
				which = rce_monthly_religion_mechanic_hindu_value_other
				value = modifier:rce_monthly_religion_mechanic_hindu_change
			}
			change_variable = { which = rce_monthly_religion_mechanic_hindu_value which = rce_monthly_religion_mechanic_hindu_value_other }
			# Level of Dedication
			if = {
				limit = {
					rce_has_religion_mechanic_hindu_major_deity_dedication_level = { major_deity = current level = 5 }
				}
				set_variable = { which = rce_monthly_religion_mechanic_hindu_value_dedication_level value = -0.09 }
			}
			else_if = {
				limit = {
					rce_has_religion_mechanic_hindu_major_deity_dedication_level = { major_deity = current level = 4 }
				}
				set_variable = { which = rce_monthly_religion_mechanic_hindu_value_dedication_level value = -0.07 }
			}
			else_if = {
				limit = {
					rce_has_religion_mechanic_hindu_major_deity_dedication_level = { major_deity = current level = 3 }
				}
				set_variable = { which = rce_monthly_religion_mechanic_hindu_value_dedication_level value = -0.05 }
			}
			else_if = {
				limit = {
					rce_has_religion_mechanic_hindu_major_deity_dedication_level = { major_deity = current level = 2 }
				}
				set_variable = { which = rce_monthly_religion_mechanic_hindu_value_dedication_level value = -0.03 }
			}
			else_if = {
				limit = {
					rce_has_religion_mechanic_hindu_major_deity_dedication_level = { major_deity = current level = 1 }
				}
				set_variable = { which = rce_monthly_religion_mechanic_hindu_value_dedication_level value = -0.01 }
			}
			else = {
				set_variable = { which = rce_monthly_religion_mechanic_hindu_value_dedication_level value = 0 }
			}
			change_variable = { which = rce_monthly_religion_mechanic_hindu_value which = rce_monthly_religion_mechanic_hindu_value_dedication_level }
			# Religious Unity
			if = {
				limit = {
					NOT = { religious_unity = 0.25 }
				}
				set_variable = { which = rce_monthly_religion_mechanic_hindu_value_religious_unity value = -0.03 }
			}
			else_if = {
				limit = {
					NOT = { religious_unity = 0.5 }
				}
				set_variable = { which = rce_monthly_religion_mechanic_hindu_value_religious_unity value = -0.02 }
			}
			else_if = {
				limit = {
					NOT = { religious_unity = 0.75 }
				}
				set_variable = { which = rce_monthly_religion_mechanic_hindu_value_religious_unity value = -0.01 }
			}
			else = {
				set_variable = { which = rce_monthly_religion_mechanic_hindu_value_religious_unity value = 0 }
			}
			change_variable = { which = rce_monthly_religion_mechanic_hindu_value which = rce_monthly_religion_mechanic_hindu_value_religious_unity }

			if = {
				limit = {

						AND = {
							rce_has_religion_mechanic_hindu_current_major_deity = { name = brahma }
							ADM = 6
						}

							DIP = 6
						}
						AND = {

							MIL = 6
						}
					}

					limit = {

				}
				else = {

				}
