rce_release_celtic_vassal_subject = {
	rce_release_celtic_vassal_subject_culture_$culture$ = yes
}
rce_release_celtic_vassal_subject_in_target = {
	tooltip = {
		release = $who$
		$who$ = {
			override_country_name = RCE_CELTIC_VASSAL_$culture$
		}
		create_subject = {
			subject = $who$
			subject_type = rce_celtic_vassal
		}
		$who$ = {
			change_primary_culture = $culture$
		}
		custom_tooltip = rce_release_celtic_vassal_subject_in_target_tt
		every_owned_province = {
			limit = {
				is_capital = no
				rce_province_is_in_region_to_celtic_vassal = { culture = $culture$ }
			}
			custom_tooltip = rce_release_celtic_vassal_subject_in_target_tt_list
		}
	}
	hidden_effect = {
		every_owned_province = {
			limit = {
				is_core = $who$
			}
			set_province_flag = rce_was_core_of_@$who$
			remove_core = $who$
		}
		every_owned_province = {
			limit = {
				is_capital = no
				rce_province_is_in_region_to_celtic_vassal = { culture = $culture$ }
			}
			add_core = $who$
			remove_core = PREV
		}
		release = $who$
		every_owned_province = {
			limit = {
				has_province_flag = rce_was_core_of_@$who$
			}
			clr_province_flag = rce_was_core_of_@$who$
			add_core = $who$
		}
		$who$ = {
			change_primary_culture = $culture$
			override_country_name = RCE_CELTIC_VASSAL_$culture$
			change_country_color = { country = PREV }
			if = {
				limit = {
					is_lacking_institutions = yes
				}
				embrace_institutions_of = PREV
			}
			if = {
				limit = {
					PREV = {
						government = monarchy
					}
				}
				change_government = monarchy
			}
			else_if = {
				limit = {
					PREV = {
						government = republic
					}
				}
				change_government = republic
			}
			else_if = {
				limit = {
					PREV = {
						government = theocracy
					}
				}
				change_government = theocracy
			}
			else_if = {
				limit = {
					PREV = {
						government = tribal
					}
				}
				change_government = tribal
			}
			else_if = {
				limit = {
					PREV = {
						government = native
					}
				}
				change_government = native
			}
			else = {
				change_government = monarchy
			}
			adopt_reform_progress = PREV
			change_religion = PREV
			define_ruler = {
				max_age = 60
				claim = 100
			}
		}
		create_subject = {
			subject = $who$
			subject_type = rce_celtic_vassal
		}
	}
}
