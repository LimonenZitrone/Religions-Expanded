rce_set_religion_mechanic_druidic_council_value = {
	if = {
		limit = {
			rce_uses_religion_mechanic = { mechanic = druidic_council }
		}
		[[value]
			set_variable = { which = rce_religion_mechanic_druidic_council_value value = $value$ }
		]
		[[which]
			set_variable = { which = rce_religion_mechanic_druidic_council_value which = $which$ }
		]
		hidden_effect = {
			if = {
				limit = {
					rce_has_religion_mechanic_druidic_council_value = { value = 100 }
				}
				set_variable = { which = rce_religion_mechanic_druidic_council_value value = 100 }
			}
			else_if = {
				limit = {
					NOT = { rce_has_religion_mechanic_druidic_council_value = { value = -99.99 } }
				}
				set_variable = { which = rce_religion_mechanic_druidic_council_value value = -100 }
			}
			set_variable = { which = rce_religion_mechanic_druidic_council_value_display which = rce_religion_mechanic_druidic_council_value }
			divide_variable = { which = rce_religion_mechanic_druidic_council_value_display value = 1000 }
			multiply_variable = { which = rce_religion_mechanic_druidic_council_value_display value = 1000 } # 0 decimals in display
			set_variable = { which = rce_religion_mechanic_druidic_council_value_progressbar which = rce_religion_mechanic_druidic_council_value }
			change_variable = { which = rce_religion_mechanic_druidic_council_value_progressbar value = 100 }
			divide_variable = { which = rce_religion_mechanic_druidic_council_value_progressbar value = 2 } # 0-100%
			rce_update_religion_mechanic_druidic_council_modifier = yes
		}
	}
}
